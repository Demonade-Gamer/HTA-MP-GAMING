		has_country_flag = alaska_purchase_interest
	}

		months = 6
		modifier = {
			factor = 0.8
			tag = SKM
		}
		modifier = {
			factor = 0.8
			alliance_with = HIS
		}
		modifier = {
			factor = 0.8
			relation = { who = HIS value = 100 }
		}
		modifier = {
			factor = 1.2
			NOT = { relation = { who = HIS value = 50 } }
		}
		modifier = {
			factor = 0.1
			year = 1870
		}
	}
	
	immediate = {
		set_global_flag = alaska_purchase_process_global
	}
	
	option = {
		name = "Make the offer."
		treasury = -250000
		HIS = { country_event = 32510 }
		ai_chance = {
			factor = 95
			modifier = {

			}
		}
	}
	
	option = {
		name = "We don't need it."
		clr_country_flag = alaska_purchase_interest
		set_country_flag = refused_alaska
		ai_chance = {
			factor = 5
			modifier = {
				factor = 2
				NOT = { money = 100000 }
			}
			modifier = {
				factor = 0
				OR = {
					tag = SKM
					tag = QUE
				}
			}
		}
	}
}

country_event = {
	id = 32510
	title = "EVTNAME32510_NEW" #the Alaskan Purchase
	desc = "EVTDESC32510_NEW"
	picture = "alaska"
	
	is_triggered_only = yes
	
	option = {
		name = "Sell it."
		set_global_flag = alaska_purchase_completed
		prestige = 10
		treasury = 250000
		USA_1 = { remove_core = HIS }
		FROM = { country_event = 32511 }
		ai_chance = {
			factor = 75
		}
	}
	
	option = {
		name = "Refuse, but find another buyer."
		clr_global_flag = alaska_purchase_process_global
		FROM = {
			clr_country_flag = alaska_purchase_interest
			country_event = 32514
		}
		any_country = {
			limit = {
				NOT = { tag = HIS }
				NOT = { has_country_flag = refused_alaska }
				OR = {
					is_greater_power = yes
					is_secondary_power = yes
				}
				any_owned_province = {
					OR = {
						region = MEX_85
						region = USA_78
						region = USA_81
						region = CAN_13
						region = JAP_1088
					}
				}					
			}
			set_country_flag = alaska_purchase_interest
		}
		ai_chance = {
			factor = 25
			modifier = {
				factor = 0
				FROM = {
					OR = {
						tag = SKM
						tag = QUE
					}
				}
			}
		}
	}
	
	option = {
		name = "We have changed our mind. Alaska is not for sale."
		prestige = -10
		USA_1 = { add_core = HIS }
		ai_chance = {
			factor = 0
		}
	}
}
