			factor = 75
			modifier = {
				factor = 0
				badboy = 0.88
			}
			modifier = {
				factor = 100
				has_global_flag = railroading_on
			}
		}
	}
	
	option = {
		name = "This is not in our best interests now."
		prestige = -2

		}
