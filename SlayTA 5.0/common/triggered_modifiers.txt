ai_boost = {
	trigger = { 
		ai = yes
			
		civilized = yes
		NOT = { literacy = 0.8 }
		NOT = {
		 	is_culture_group = east_asian
			OR = { 
				has_pop_culture = nanfaren
				has_pop_culture = yue 
			}
		}
	}
	tax_efficiency = 0.05 #5% increase
	education_efficiency_modifier = 1.0 #100% boost
 	loan_interest = -0.03 #3% less interest on debts
	icon = 9
}
